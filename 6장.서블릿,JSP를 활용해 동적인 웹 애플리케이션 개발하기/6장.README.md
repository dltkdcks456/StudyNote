### ✅ Project를 Maven → Gradle로 변경 시 의존성 항목

```groovy
dependencies {
	implementation 'org.springframework.boot:spring-boot-starter-data-jdbc'
	implementation 'org.springframework.boot:spring-boot-starter-web'
	compileOnly 'org.projectlombok:lombok'
	runtimeOnly 'com.h2database:h2'
	annotationProcessor 'org.projectlombok:lombok'
	testImplementation 'org.springframework.boot:spring-boot-starter-test'

	// Servlet API
    // Java EE웹 애플리케이션을 개발할 때 사용하는 기본 인터페이스와 클래스를 제공한다.
	implementation 'javax.servlet:javax.servlet-api:3.1.0'
    
    // JSTL(JavaServer Pages Standard Tag Library)
    // JSP(JavaServerPages)에서 커스텀 태그를 사용하여 공통 작업을 쉽게 처리할 수 있게 도와준다.
	implementation 'javax.servlet:jstl:1.2'

	// Google Guava
    // Guava 라이브러리는 자바의 코어 라이브러리를 보완하는 유틸리티 컬렉션, 캐싱, 프리미티브 지원, 
    // 동시성 라이브러리, 공통 주석, 문자열 처리 등을 제공한다.
	implementation 'com.google.guava:guava:18.0'
	
	// Apache Commons
    // 자바의 기본 java.lang 패키지를 보완하는 추가 유틸리티를 제공한다. 문자열 조작, 숫자 조작,
    // 시스템 속성 접근 등의 기능을 포함한다.
	implementation 'org.apache.commons:commons-lang3:3.3.2'
    
    // Apache Commons Database Connection Pooling2는 데이터베이스 연결 풀링 서비스를 제공한다.
    // 데이터베이스 연결을 효율적으로 재사용하여 성능을 향상시킨다.
	implementation 'org.apache.commons:commons-dbcp2:2.1.1'
    
	// Reflections
    // 런타임에 클래스, 필드, 메서드, 어노테이션 등을 스캔하여 작업을 수행할 수 있게 도와주는 유틸리티
	implementation 'org.reflections:reflections:0.9.10'

	// Logback
	// 로깅을 위한 강력하고 유연한 프레임워크
    // SLF4J의 구현체이며 logback-core는 logback-classic의 기반이 된다.
	testImplementation 'ch.qos.logback:logback-classic:1.4.14'
	implementation 'ch.qos.logback:logback-core:1.4.14'

	// Jackson
    // JSON 데이터를 Java 객체로 변환하거나, Java 객체를 JSON으로 변환할 때 사용
	implementation 'com.fasterxml.jackson.core:jackson-databind:2.4.4'

	// Tomcat Embed
	// 임베디드 Tomcat 서버를 설정하고, JSP지원(tomcat-embed-jasper) 
    // 및 Tomcat의 JULI로깅(tomcat-embed-logging-juli)을 활용할 수 있다.
	implementation 'org.apache.tomcat.embed:tomcat-embed-core:8.0.15'
	implementation 'org.apache.tomcat.embed:tomcat-embed-logging-juli:8.0.15'
	implementation 'org.apache.tomcat.embed:tomcat-embed-jasper:8.0.15'
	
    // Junit4
    // 단위 테스트 프레임워크
	testImplementation 'junit:junit:4.12'
}
```
